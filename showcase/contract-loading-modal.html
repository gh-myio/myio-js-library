Implementation Example

How to use the contract loading modal with detailed breakdown.

// Store contract state globally for HEADER access window.CONTRACT_STATE = { isLoaded: false, isValid: false, energy: { total: 0, entries: 0, commonArea: 0, stores: 0 }, water: { total: 0, entries: 0, commonArea: 0, stores: 0 }, temperature: { total: 0, internal: 0, stores: 0 } }; // Update domain with detailed counts function updateContractModalDomain(domain, counts) { const section = document.getElementById(`section${domain}`); // Update totals and details if (domain === 'energy') { window.CONTRACT_STATE.energy = counts; document.getElementById('totalEnergy').textContent = `${counts.total} devices`; document.getElementById('energyEntries').textContent = counts.entries; document.getElementById('energyCommonArea').textContent = counts.commonArea; document.getElementById('energyStores').textContent = counts.stores; } // ... similar for water and temperature section.classList.add('loaded'); }
