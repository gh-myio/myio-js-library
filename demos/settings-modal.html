ðŸ“š ThingsBoard Integration Example

Example of how to integrate with a real ThingsBoard widget.

// In ThingsBoard widget controller function openDeviceSettings() { const jwtToken = localStorage.getItem('jwt_token'); const deviceId = self.ctx.datasources[0]?.entityId; if (!jwtToken) { alert('Please log in to modify settings'); return; } MyIOLibrary.openDashboardPopupSettings({ deviceId: deviceId, jwtToken: jwtToken, onSaved: (result) => { if (result.ok) { // Refresh widget data self.ctx.detectChanges(); console.log('Settings updated successfully'); } else { console.error('Failed to save settings:', result); } }, onError: (error) => { if (error.userAction === 'RE_AUTH') { // Redirect to login or show auth modal window.location.href = '/login'; } } }); }
